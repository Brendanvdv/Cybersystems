# Program to bubble sort an array of 10 values
# Check the expected output and compare it with the one of your simulator
# The format is: <instruction mem. address>: <OPCODE> <COMMA SEPARATED OPERANDS>;
# All numers are decimal 

# Copy the array from source (0-9) to destination (10-19)
0: LI R1, 0;  # R1 is the read pointer
1: LI R2, 0;  # R2 is the counter
2: LI R3, 2;  # R3 is what we want compare to
3: LI R4, 10;  # R4 is the counter for copy
4: LI R6, 1; # register which will be added
5: LI R13, 13;
# R5 is the temporary buffer for copy
6: LD R5, R1;  # Copy: address**
7: ADD R1, R1, R6;  # Done copy address**

8: JEQ R10, R5,R3;  # Read form mem(read pointer)
9: JEQ R13, R1,R4;  # Write to mem(write pointer)
10: JR R5;  # Go to IS EQUAL

11: ADD R2, R2, R6;  # Decrement counter**
12: JEQ R13, R1,R4;  # Increment read pointer**
13: JR R6;  # Increment write pointer**

14: END;
